CREATE TABLE LOTERIA(
ID INTEGER PRIMARY KEY,
DESCRICAO TEXT,
QTDEMINMARCACAO INTEGER,
QTDEMAXMARCACAO INTEGER)

CREATE TABLE PREMIO(
ID INTEGER PRIMARY KEY,
QTDEACERTO INTEGER,
NIVEL INTEGER,
ID_LOTERIA INTEGER,
FOREIGN KEY(ID_LOTERIA) REFERENCES LOTERIA(ID))

CREATE TABLE JOGO(
ID INTEGER PRIMARY KEY,
DESCRICAO TEXT,
JOGOPERMANENTE INTEGER,
ID_LOTERIA INTEGER,
FOREIGN KEY(ID_LOTERIA) REFERENCES LOTERIA(ID))

CREATE TABLE NUMEROJOGADO(
ID INTEGER PRIMARY KEY,
NUMERO INTEGER,
ID_JOGO INTEGER,
FOREIGN KEY (ID_JOGO) REFERENCES JOGO(ID))

CREATE TABLE CONCURSO(
ID INTEGER PRIMARY KEY,
NUMERO INTEGER,
DATA INTEGER,
ID_LOTERIA INTEGER,
FOREIGN KEY (ID_LOTERIA) REFERENCES LOTERIA(ID))

CREATE TABLE SORTEIO(
ID INTEGER PRIMARY KEY,
NUMEROSORTEIO INTEGER,
ID_CONCURSO INTEGER,
FOREIGN KEY (ID_CONCURSO) REFERENCES CONCURSO(ID))

CREATE TABLE NUMEROSORTEADO(
ID INTEGER PRIMARY KEY,
NUMERO INTEGER,
ID_SORTEIO INTEGER,
FOREIGN KEY (ID_SORTEIO) REFERENCES SORTEIO(ID))

CREATE TABLE JOGOSORTEIO(
ID_JOGO INTEGER,
ID_SORTEIO INTEGER,
ID_PREMIO INTEGER,
QTDEACERTO INTEGER,
FOREIGN KEY (ID_JOGO) REFERENCES JOGO(ID),
FOREIGN KEY (ID_SORTEIO) REFERENCES SORTEIO(ID),
FOREIGN KEY (ID_PREMIO) REFERENCES PREMIO(ID))

CREATE TABLE JOGOCONCURSO(
ID_JOGO INTEGER,
ID_CONCURSO INTEGER,
CONFERIDO INTEGER,
FOREIGN KEY (ID_JOGO) REFERENCES JOGO(ID),
FOREIGN KEY (ID_CONCURSO) REFERENCES CONCURSO(ID))
